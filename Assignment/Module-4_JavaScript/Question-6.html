6. Write the code for adding new elements dynamically?

Answer:-

html
                            <!DOCTYPE html>
                            <html lang="en">
                            <head>
                                <meta charset="UTF-8">
                                <meta name="viewport" content="width=device-width, initial-scale=1.0">
                                <link rel="stylesheet" href="CSS/Question-6.css">
                                <title>Question-6>/title>
                            </head>
                            <body>
                            <h2>adding Content</h2>
                            <div id="addContent">
                                <p>This is an adding paragraph.</p>
                            </div>
                            <button onclick="addNewParagraph()">Add New Paragraph</button>
        
                            </body>
JavaScript
                        function addNewParagraph() {
   
                            var newParagraph = document.createElement('p');
                            newParagraph.textContent = 'This is a new paragraph added dynamically.';

                            var addContent = document.getElementById('addContent');
                            addContent.appendChild(newParagraph);
                     }
Output:-

adding Content

This is an adding paragraph.
